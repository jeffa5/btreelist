pub struct BTreeList<T, const B: usize = 6> { /* private fields */ }
Expand description

A list with efficient insert and removal in the middle.

It may be worth benchmarking your use case and trying to use a Box<T> instead of a plain T as this can improve performance in some cases. Similar word-length wrapper types would also work e.g. Rc.

let mut list = BTreeList::default();
list.push(1);
list.push(2);

assert_eq!(list.len(), 2);
assert_eq!(list[0], 1);

assert_eq!(list.pop(), Some(2));
assert_eq!(list.len(), 1);

list[0] = 7;
assert_eq!(list[0], 7);

list.extend([1,2,3]);

for x in &list {
    println!("{x}");
}
assert_eq!(list, btreelist![7, 1, 2, 3]);
